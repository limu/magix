1 /**
  2  * @fileOverview 多播事件对象
  3  * @author 行列<xinglie.lkf@taobao.com>
  4  * @version 1.0
  5  **/
  6 KISSY.add("magix/event",function(S,Magix){
  7 	/**
  8  * 根据名称生成事件数组的key
  9  * @param  {Strig} name 事件名称
 10  * @return {String} 包装后的key
 11  */
 12 var genKey=function(name){
 13 	return '~~'+name+'_list'
 14 };
 15 
 16 var safeExec=Magix.safeExec;
 17 /**
 18  * 多播事件对象
 19  * @name Event
 20  * @namespace
 21  */
 22 var Event={
 23 	/**
 24 	 * @lends Event
 25 	 */
 26 	/**
 27 	 * 触发事件
 28 	 * @param {String} name 事件名称
 29 	 * @param {Object} data 事件对象
 30 	 * @param {Boolean} remove 事件触发完成后是否移除这个事件的所有监听
 31 	 * @param {Boolean} lastToFirst 是否从后向前触发事件的监听列表
 32 	 */
 33 	trigger:function(name,data,remove,lastToFirst){
 34 		var key=genKey(name),
 35 			me=this,
 36 			list=me[key];
 37 		if(list){
 38 			if(!data)data={};
 39 			if(!data.type)data.type=name;
 40 			if(lastToFirst){
 41 				for(var i=list.length-1;i>=0;i--){
 42 					if(safeExec(list[i],data,me)===false){
 43 						break;
 44 					}
 45 				}
 46 			}else{
 47 				for(var i=0,j=list.length;i<j;i++){
 48 					if(safeExec(list[i],data,me)===false){
 49 						break;
 50 					}
 51 				}
 52 			}
 53 		}
 54 		if(remove){
 55 			delete me[key];
 56 		}
 57 	},
 58 	/**
 59 	 * 绑定事件
 60 	 * @param  {String}   name 事件名称
 61 	 * @param  {Function} fn   事件回调
 62 	 */
 63 	bind:function(name,fn){
 64 		var key=genKey(name);
 65 		if(!this[key])this[key]=[];
 66 		this[key].push(fn);
 67 	},
 68 	/**
 69 	 * 解除事件绑定
 70 	 * @param  {String}   name 事件名称
 71 	 * @param  {Function} fn   事件回调
 72 	 */
 73 	unbind:function(name,fn){
 74 		var key=genKey(name),
 75 			list=this[key];
 76 		if(list){
 77 			if(fn){
 78 				for(var i=0,j=list.length;i<j;i++){
 79 					if(list[i]==fn){
 80 						list.splice(i,1);
 81 						break;
 82 					}
 83 				}
 84 			}else{
 85 				delete this[key];
 86 			}
 87 		}
 88 	}
 89 };
 90 	return Event;
 91 },{
 92 	requires:["magix/magix"]
 93 });
