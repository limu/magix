814     var ProcessObject=function(props,proto,enterObject){
815         for(var p in proto){
816             if(S.isObject(proto[p])){
817                 if(!Has(props,p))props[p]={};
818                 ProcessObject(props[p],proto[p],1);
819             }else if(enterObject){
820                 props[p]=proto[p];
821             }
822         }

826     View.prototype.fetchTmpl=function(fn){
827         var me=this;
828         var tmpl=me.template;
829         if(S.isUndefined(tmpl)){
830             var i=Magix.tmpl(me.path);
831             if(i.h){
832                 fn(i.v);
833             }else{
834                 var file=MxConfig.appHome+me.path+'.html';
835                 var l=ProcessObject[file];
836                 var onload=function(tmpl){
837                     fn(Magix.tmpl(me.path,tmpl));
838                 };
839                 if(l){
840                     l.push(onload);
841                 }else{
842                     l=ProcessObject[file]=[onload];
843                     IO({
844                         url:file+(MxConfig.debug?'?t='+S.now():''),
845                         success:function(x){
846                             SafeExec(l,x);
847                             delete ProcessObject[file];
848                         },
849                         error:function(e,m){
850                             SafeExec(l,m);
851                             delete ProcessObject[file];
852                         }
853                     });
854                 }
855             }
856         }else{
857             fn(tmpl);
858         }
