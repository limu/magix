1 /**
  2  * @fileOverview VOM
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add("magix/vom",function(S,Vframe,Magix,Event,Body){
  7     var Has=Magix.has;
  8 var VframesCount=0;
  9 var FirstVframesLoaded=0;
 10 var LastPercent=0;
 11 var FirstReady=0;
 12 var Vframes={};
 13 var Loc;
 14 
 15 /**
 16  * VOM对象
 17  * @name VOM
 18  * @namespace
 19  */
 20 var VOM=Magix.mix({
 21     /**
 22      * @lends VOM
 23      */
 24     /**
 25      * 获取所有的vframe对象
 26      * @return {Object}
 27      */
 28     all:function(){
 29         return Vframes;
 30     },
 31     /**
 32      * 注册vframe对象
 33      * @param {Vframe} vf Vframe对象
 34      */
 35     add:function(vf){
 36         if(!Has(Vframes,vf.id)){
 37             VframesCount++;
 38             Vframes[vf.id]=vf;
 39             vf.owner=VOM;
 40             VOM.fire('add',{vframe:vf});
 41         }
 42     },
 43     /**
 44      * 根据vframe的id获取vframe对象
 45      * @param {String} id vframe的id
 46      * @return {Vframe} vframe对象
 47      */
 48     get:function(id){
 49         return Vframes[id];
 50     },
 51     /**
 52      * 删除已注册的vframe对象
 53      * @param {String} id vframe对象的id
 54      */
 55     remove:function(id){
 56         //var id=Magix.isString(vf)?vf:vf.id;
 57         var vf=Vframes[id];
 58         if(vf){
 59             VframesCount--;
 60             if(vf.fcc)FirstVframesLoaded--;
 61             delete Vframes[id];
 62             VOM.fire('remove',{vframe:vf});
 63         }        
 64     },
 65     /**
 66      * 通知其中的一个view创建完成
 67      */
 68     childCreated:function(){
 69         if(!FirstReady){
 70             FirstVframesLoaded++;
 71             var np=FirstVframesLoaded/VframesCount;
 72             if(LastPercent<np){
 73                 VOM.fire('progress',{
 74                     percent:LastPercent=np
 75                 });
 76                 if(np==1){
 77                     FirstReady=1;
 78                     VOM.un('progress');
 79                 }
 80             }
 81         }
 82     },
 83     /**
 84      * 获取根vframe对象
 85      */
 86     root:function(){
 87         return Vframe.root(VOM);
 88     },
 89     /**
 90      * 重新渲染根vframe
 91      * @param {Object} e Router.locationChanged事件对象
 92      * @param {Object} e.location window.location.href解析出来的对象
 93      * @param {Object} e.changed 包含有哪些变化的对象
 94      */
 95     remountRoot:function(e){
 96         //
 97         var vf=VOM.root();
 98         //me.$loc=e.location;
 99         //
100         Loc=e.location;
101         vf.mountView(Loc.view);
102     },
103     /**
104      * 向vframe通知地址栏发生变化
105      * @param {Object} e 事件对象
106      * @param {Object} e.location window.location.href解析出来的对象
107      * @param {Object} e.changed 包含有哪些变化的对象
108      */
109     locationChanged:function(e){
110         Loc=e.location;
111         var vf=VOM.root();
112         vf.locationChanged(Loc,e.changed);
113     },
114     /**
115      * 更新view的location对象
116      * @param  {Object} loc location
117      */
118     locationUpdated:function(loc){
119         Loc=loc;
120         var vf=VOM.root();
121         vf.locationUpdated(loc);
122     },
123     /**
124      * 获取window.location.href解析后的对象
125      * @return {Object}
126      */
127     getLocation:function(){
128         return Loc;
129     }
130     /**
131      * view加载完成进度
132      * @name VOM.progress
133      * @event
134      * @param {Object} e
135      * @param {Object} e.precent 百分比
136      */
137     /**
138      * 注册vframe对象时触发
139      * @name VOM.add
140      * @event
141      * @param {Object} e
142      * @param {Vframe} e.vframe
143      */
144     /**
145      * 删除vframe对象时触发
146      * @name VOM.remove
147      * @event
148      * @param {Object} e
149      * @param {Vframe} e.vframe
150      */
151 },Event);
152     Body.VOM=VOM;
153     Body.on('event',function(e){
154         var vframe=VOM.get(e.hld);
155         var view=vframe&&vframe.view;
156         if(view){
157             view.processEvent(e);
158         }
159     });
160     return VOM;
161 },{
162     requires:["magix/vframe","magix/magix","magix/event","magix/body"]
163 });
