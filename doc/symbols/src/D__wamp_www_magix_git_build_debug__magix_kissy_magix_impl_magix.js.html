1 /**
  2  * @fileOverview magix中根据底层类库需要重写的方法
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add('magix/impl/magix',function(S){
  7 	return {
  8 		include:function(path){
  9 			var url = S.Config.packages.magix.path+path + ".js?r=" + Math.random()+'.js';
 10 			var xhr = window.ActiveXObject || window.XMLHttpRequest;
 11 			var r = new xhr('Microsoft.XMLHTTP');
 12 			r.open('GET', url, false);
 13 			r.send(null);
 14 			return r.responseText;
 15 		},
 16 		libRequire:function(name,fn){
 17 			S.use(name,function(S,T){
 18 				fn(T)
 19 			});
 20 		},
 21 		libEnv:function(){
 22 			var me=this;
 23 			var cfg=me.config();
 24 			var appHome=cfg.appHome;
 25 			if(!appHome){
 26 				throw new Error('please set appHome');
 27 			}
 28 			appHome=appHome.replace(/(^|\/)app\/?/i,function(a,b){
 29 				return b||'./';
 30 			});
 31 			cfg.appHome=appHome;
 32 			
 33 			if(!cfg.release&&/^https?:\/\//.test(appHome)){
 34 				cfg.release= appHome.indexOf(location.protocol+'//'+location.host)==-1;
 35 			}
 36 			if(!cfg.release){
 37 				var reg=new RegExp("("+appHome+".+)-min\\.js(\\?[^?]+)?");
 38 				S.config({
 39 					map:[[reg,'$1.js$2']]
 40 				});
 41 			}
 42 			var appTag='';
 43 			if(cfg.release){
 44 				appTag=cfg.appTag;
 45 			}else{
 46 				appTag=S.now();
 47 			}
 48 			if(appTag){
 49 				appTag+='.js';
 50 			}
 51 			S.config({
 52 				packages:[{
 53 					name:'app',
 54 					path:appHome,
 55 					tag:appTag
 56 				}]
 57 			});
 58 			var vfanim='mxext/vfanim';
 59 			if(cfg.viewChangeAnim&&S.Env.mods[vfanim]){
 60 				S.use(vfanim);
 61 			}
 62 		},
 63 		isArray:S.isArray,
 64 		isFunction:S.isFunction,
 65 		isObject:S.isObject,
 66 		isRegExp:S.isRegExp,
 67 		isString:S.isString,
 68 		isNumber:S.isNumber
 69 	}
 70 });
