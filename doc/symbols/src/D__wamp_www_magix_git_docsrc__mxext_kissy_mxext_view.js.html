  9     var DestroyManagedTryList='destroy,abort,stop,cancel,remove'.split(COMMA);
 10     var ResCounter=0;
 11     var safeExec=Magix.safeExec;
 12     var HAS=Magix.has;
 13     var VOMEventsObject={};
 14     var PrepareVOMMessage=function(vom){
 15         if(!PrepareVOMMessage.d){
 16             PrepareVOMMessage.d=1;
 17             vom.on('add',function(e){
 18                 var vf=e.vframe;
 19                 var list=VOMEventsObject[vf.id];
 20                 if(list){
 21                     for(var i=0;i<list.length;i++){
 22                         PostMessage(vf,list[i]);
 23                     }
 24                     delete VOMEventsObject[vf.id];
 25                 }
 26             });
 27             vom.on('remove',function(e){
 28                 delete VOMEventsObject[e.vframe.id];
 29             });
 30             var vf=vom.root();
 31             vf.on('childrenCreated',function(){
 32                 VOMEventsObject={};
 33             });
 34         }
 35     };
 36     var PostMessage=function(vframe,args){
 37         var view=vframe.view;
 38         if(view&&vframe.viewUsable){
 39             safeExec(view.receiveMessage,args,view);
 40         }else{
 41             var interact=function(e){
 42                 vframe.un('viewInteract',interact);
 43                 safeExec(e.view.receiveMessage,args,e.view);
 44             };
 45             vframe.on('viewInteract',interact);
 46         }
 47     };

 54     var MxView=View.extend({
 55         mxViewCtor:Magix.noop,//供扩展用
 56         /**
 57          * 调用magix/router的navigate方法
 58          * @param {Object|String} params 参数字符串或参数对象
 59          */
 60         navigate:function(params){
 61             Router.navigate.apply(Router,arguments);
 62         },
 63         /**
 64          * 让view帮你管理资源，对于异步回调函数更应该调用该方法进行托管
 65          * 当调用该方法后，您不需要在异步回调方法内判断当前view是否已经销毁
 66          * 同时对于view刷新后，上个异步请求返回刷新界面的问题也得到很好的解决。<b>强烈建议对异步回调函数，组件等进行托管</b>
 67          * @param {String|Object} key 托管的资源或要共享的资源标识key
 68          * @param {Object} [res] 要托管的资源
 69          * @return {Object} 返回传入的资源，对于函数会自动进行一次包装
 70          * @example
 71          * init:function(){
 72          *      this.manage('user_list',[//管理对象资源
 73          *          {id:1,name:'a'},
 74          *          {id:2,name:'b'}
 75          *      ]);
 76          * },
 77          * render:function(){
 78          *      var _self=this;
 79          *      var m=new Model();
 80          *      m.load({
 81          *          success:_self.manage(function(resp){//管理匿名函数
 82          *              //TODO
 83          *              var brix=new BrixDropdownList();
 84          *
 85          *              _self.manage(brix);//管理组件
 86          *
 87          *              var pagination=_self.manage(new BrixPagination());//也可以这样
 88          *
 89          *              var timer=_self.manage(setTimeout(function(){
 90          *                  S.log('timer');
 91          *              },2000));//也可以管理定时器
 92          *
 93          *              
 94          *              var userList=_self.getManaged('user_list');//通过key取托管的资源
 95          *
 96          *              S.log(userList);
 97          *          }),
 98          *          error:_self.manage(function(msg){
 99          *              //TODO
100          *          })
101          *      })
102          * }
103          */
104         manage:function(key,res){
105             var me=this;
106             var args=arguments;
107             var hasKey=true;
108             if(args.length==1){
109                 res=key;
110                 key='res_'+(ResCounter++);
111                 hasKey=false;
112             }
113             if(!me.$resCache)me.$resCache={};
114             var wrapObj={
115                 hasKey:hasKey,
116                 res:res
117             };
118             me.$resCache[key]=wrapObj;
119             return res;
120         },
121         /**
122          * 获取托管的资源
123          * @param {String} key 托管资源时传入的标识key
124          * @return {[type]} [description]
125          */
126         getManaged:function(key){
127             var me=this;
128             var cache=me.$resCache;
129             var sign=me.sign;
130             if(cache&&HAS(cache,key)){
131                 var wrapObj=cache[key];
132                 var resource=wrapObj.res;
133                 return resource;
134             }
135             return null;
136         },
137         /**
138          * 移除托管的资源
139          * @param {String|Object} param 托管时标识key或托管的对象
140          * @return {Object} 返回移除的资源
141          */
142         removeManaged:function(param){
143             var me=this,res=null;
144             var cache=me.$resCache;
145             if(cache){
146                 if(HAS(cache,param)){
147                     res=cache[param].res;
148                     delete cache[param];
149                 }else{
150                     for(var p in cache){
151                         if(cache[p].res===param){
152                             res=cache[p].res;
153                             delete cache[p];
154                             break;
155                         }
156                     }
157                 }
158             }
159             return res;
160         },
161         /**
162          * 销毁托管的资源
163          * @param {Boolean} [byRefresh] 是否是刷新时的销毁
164          * @private
165          */
166         destroyManaged:function(byRefresh){
167             var me=this;
168             var cache=me.$resCache;
169             //
170             if(cache){
171                 for(var p in cache){
172                     var o=cache[p];
173                     //var processed=false;
174                     var res=o.res;
175                     if(Magix.isNumber(res)){//数字，有可能是定时器
176                         WIN.clearTimeout(res);
177                         WIN.clearInterval(res);
178                         //processed=true;
179                     }else if(res){
180                         if(res.nodeType&&res.parentNode){
181                             S.one(res).remove();
182                             //processed=true;
183                         }else{
184                             for(var i=0;i<DestroyManagedTryList.length;i++){
185                                 if(Magix.isFunction(res[DestroyManagedTryList[i]])){
186                                     safeExec(res[DestroyManagedTryList[i]],[],res);
187                                     //processed=true;
188                                     //不进行break,比如有时候可能存在abort 和  destroy
189                                 }
190                             }
191                         }
192                     }
193                     /*me.fire('destroyResource',{
194                         resource:res,
195                         processed:processed
196                     });*/
197                     if(byRefresh&&!o.hasKey){//如果是刷新且托管时没有给key值，则表示这是一个不会在其它方法内共享托管的资源，view刷新时可以删除掉
198                         delete cache[p];
199                     }
200                 }
201                 if(!byRefresh){//如果不是刷新，则是view的销毁
202                     //me.un('destroyResource');
203                     delete me.$resCache;
204                 }
205             }
206         },
207         /**
208          * 用于接受其它view通过postMessageTo方法发来的消息，供最终的view开发人员进行覆盖
209          * @function
210          * @param {Object} e 通过postMessageTo传递的第二个参数
211          */
212         receiveMessage:Magix.noop,
213         /**
214          * 向某个vframe发送消息
215          * @param {Array|String} aims  目标vframe id数组
216          * @param {Object} args 消息对象
217          */
218         postMessageTo:function(aims,args){
219             var vom=this.vom;
220             PrepareVOMMessage(vom);
221 
222             if(!Magix.isArray(aims)){
223                 aims=[aims];
224             }
225             if(!args)args={};
226             for(var i=0,it;i<aims.length;i++){
227                 it=aims[i];
228                 var vframe=vom.get(it);
229                 if(vframe){
230                     PostMessage(vframe,args);
231                 }else{
232                     if(!VOMEventsObject[it]){
233                         VOMEventsObject[it]=[];
234                     }
235                     VOMEventsObject[it].push(args);
236                 }
237             }
238         },
239         /**
240          * @private
241          */
242         destroyMRequest:function(){
243             var me=this;
244             var cache=me.$resCache;
245             if(cache){
246                 for(var p in cache){
247                     var o=cache[p];
248                     var res=o.res;
249                     if(res&&res.fetchOne&&res.fetchAll){//销毁MRequest
250                         res.destroy();
251                         delete cache[p];
252                     }
253                 }
254             }
255         }
256     },function(){
257         var me=this;
258         me.on('interact',function(){
259             me.on('rendercall',function(){
260                 me.destroyMRequest();
261             });
262             me.on('prerender',function(){
263                 me.destroyManaged(true);
264             });
265             me.on('destroy',function(){
266                 me.destroyManaged();
267             });
268         });
269         me.mxViewCtor();
270     });
