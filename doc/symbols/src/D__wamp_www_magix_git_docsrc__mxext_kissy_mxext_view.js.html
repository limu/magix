 18     var ResCounter=0;

 20     var Has=Magix.has;
 21     var VOMEventsObject={};
 22     var PrepareVOMMessage=function(vom){
 23         if(!PrepareVOMMessage.d){
 24             PrepareVOMMessage.d=1;
 25             vom.on('add',function(e){
 26                 var vf=e.vframe;
 27                 var list=VOMEventsObject[vf.id];
 28                 if(list){
 29                     for(var i=0;i<list.length;i++){
 30                         PostMessage(vf,list[i]);
 31                     }
 32                     delete VOMEventsObject[vf.id];
 33                 }
 34             });
 35             vom.on('remove',function(e){
 36                 delete VOMEventsObject[e.vframe.id];
 37             });
 38             var vf=vom.root();
 39             vf.on('created',function(){
 40                 VOMEventsObject={};
 41             });
 42         }
 43     };
 44     var PostMessage=function(vframe,args){
 45         var view=vframe.view;
 46         if(view&&vframe.viewInited){
 47             SafeExec(view.receiveMessage,args,view);
 48         }else{
 49             var interact=function(e){
 50                 vframe.un('viewInited',interact);
 51                 SafeExec(e.view.receiveMessage,args,e.view);
 52             };
 53             vframe.on('viewInited',interact);
 54         }
 55     };

 62     var MxView=View.extend({

 65          */
 66         /**
 67          * 当前view实例化后调用，供子类重写
 68          * @function
 69          */
 70         mxViewCtor:Magix.noop,//供扩展用
 71         /**

 74         navigate:function(){
 75             Router.navigate.apply(Router,arguments);
 76         },
 77         /**
 78          * 让view帮你管理资源，<b>强烈建议对组件等进行托管</b>
 79          * @param {String|Object} key 托管的资源或要共享的资源标识key
 80          * @param {Object} res 要托管的资源
 81          * @return {Object} 返回传入的资源，对于函数会自动进行一次包装
 82          * @example
 83          * init:function(){
 84          *      this.manage('user_list',[//管理对象资源
 85          *          {id:1,name:'a'},
 86          *          {id:2,name:'b'}
 87          *      ]);
 88          * },
 89          * render:function(){
 90          *      var _self=this;
 91          *      var m=new Model();
 92          *      m.load({
 93          *          success:function(resp){
 94          *              //TODO
 95          *              var brix=new BrixDropdownList();
 96          *
 97          *              _self.manage(brix);//管理组件
 98          *
 99          *              var pagination=_self.manage(new BrixPagination());//也可以这样
100          *
101          *              var timer=_self.manage(setTimeout(function(){
102          *                  S.log('timer');
103          *              },2000));//也可以管理定时器
104          *
105          *              
106          *              var userList=_self.getManaged('user_list');//通过key取托管的资源
107          *
108          *              S.log(userList);
109          *          },
110          *          error:function(msg){
111          *              //TODO
112          *          }
113          *      });
114          *
115          *      _self.manage(m);
116          * }
117          */
118         manage:function(key,res){
119             var me=this;
120             var args=arguments;
121             var hasKey=true;
122             if(args.length==1){
123                 res=key;
124                 key='res_'+(ResCounter++);
125                 hasKey=false;
126             }
127             if(!me.$res)me.$res={};
128             var destroy;
129             if(S.isNumber(res)){
130                 destroy=DestroyTimer;
131             }else if(res&&res.destroy){
132                 destroy=Destroy;
133             }
134             var wrapObj={
135                 hasKey:hasKey,
136                 res:res,
137                 destroy:destroy
138             };
139             me.$res[key]=wrapObj;
140             return res;
141         },
142         /**
143          * 获取托管的资源
144          * @param {String} key 托管资源时传入的标识key
145          * @return {Object}
146          */
147         getManaged:function(key){
148             var me=this;
149             var cache=me.$res;
150             var sign=me.sign;
151             if(cache&&Has(cache,key)){
152                 var wrapObj=cache[key];
153                 var resource=wrapObj.res;
154                 return resource;
155             }
156             return null;
157         },
158         /**
159          * 移除托管的资源
160          * @param {String|Object} param 托管时标识key或托管的对象
161          * @return {Object} 返回移除的资源
162          */
163         removeManaged:function(param){
164             var me=this,res=null;
165             var cache=me.$res;
166             if(cache){
167                 if(Has(cache,param)){
168                     res=cache[param].res;
169                     delete cache[param];
170                 }else{
171                     for(var p in cache){
172                         if(cache[p].res===param){
173                             res=cache[p].res;
174                             delete cache[p];
175                             break;
176                         }
177                     }
178                 }
179             }
180             return res;
181         },
182         /**
183          * 销毁托管的资源
184          * @private
185          */
186         destroyManaged:function(e){
187             var me=this;
188             var cache=me.$res;
189             //
190             if(cache){
191                 for(var p in cache){
192                     var o=cache[p];
193                     //var processed=false;
194                     var res=o.res;
195                     var destroy=o.destroy;
196                     var processed=false;
197                     if(destroy){
198                         destroy(res);
199                         processed=true;
200                     }
201                     if(!o.hasKey){//如果托管时没有给key值，则表示这是一个不会在其它方法内共享托管的资源，view刷新时可以删除掉
202                         delete cache[p];
203                     }
204                     me.fire('destroyManaged',{
205                         resource:res,
206                         processed:processed
207                     });
208                 }
209                 if(e.type=='destroy'){//如果不是刷新，则是view的销毁
210                     //me.un('destroyResource');
211                     delete me.$res;
212                 }
213             }
214         },
215         /**
216          * 用于接受其它view通过postMessageTo方法发来的消息，供最终的view开发人员进行覆盖
217          * @function
218          * @param {Object} e 通过postMessageTo传递的第二个参数
219          */
220         receiveMessage:Magix.noop,
221         /**
222          * 向某个vframe发送消息
223          * @param {Array|String} aims  目标vframe id数组
224          * @param {Object} args 消息对象
225          */
226         postMessageTo:function(aims,args){
227             var vom=this.vom;
228             PrepareVOMMessage(vom);
229 
230             if(!Magix.isArray(aims)){
231                 aims=[aims];
232             }
233             if(!args)args={};
234             for(var i=0,it;i<aims.length;i++){
235                 it=aims[i];
236                 var vframe=vom.get(it);
237                 if(vframe){
238                     PostMessage(vframe,args);
239                 }else{
240                     if(!VOMEventsObject[it]){
241                         VOMEventsObject[it]=[];
242                     }
243                     VOMEventsObject[it].push(args);
244                 }
245             }
246         },
247         /**
248          * @private
249          */
250         destroyMRequest:function(){
251             var me=this;
252             var cache=me.$res;
253             if(cache){
254                 for(var p in cache){
255                     var o=cache[p];
256                     var res=o.res;
257                     if(res&&res.fetchOne&&res.fetchAll){//销毁MRequest
258                         res.destroy();
259                         delete cache[p];
260                     }
261                 }
262             }
263         }
264     },function(){
265         var me=this;
266         me.home=Magix.config().appHome;
267         me.on('interact',function(){
268             me.on('rendercall',me.destroyMRequest);
269             me.on('prerender',me.destroyManaged);
270             me.on('destroy',me.destroyManaged);
271         });
272         me.mxViewCtor();
273     });
