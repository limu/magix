151 var ReturnedSthOrEmpty=function(fn){
152 	fn=String(fn);
153 	var body=fn.match(FnBodyReg);
154 	if(body){
155 		body=body[1];
156 		if(SpaceReg.test(body))return true;
157 		body=body.replace(TransReg,EMPTY).replace(LS,EMPTY);
158 		for(var i=0,reg;i<Trims.length;i++){
159 			reg=Trims[i];
160 			while(reg.test(body)){
161 				body=body.replace(reg,EMPTY);
162 			}
163 		}
164 		return ReturnedReg.test(body);
165 	}
166 	return false;
