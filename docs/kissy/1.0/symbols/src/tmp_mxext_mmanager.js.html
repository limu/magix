 40 var DeleteCacheKey = function(modelsAttr) {
 41     if (!Magix.isArray(modelsAttr)) {
 42         modelsAttr = [modelsAttr];
 43     }
 44     for (var i = 0, m; i < modelsAttr.length; i++) {
 45         m = modelsAttr[i];
 46         delete m.cacheKey;
 47     }
 48     return modelsAttr;

 65 var Slice = [].slice;
 66 var WhiteList = {
 67     urlParams: 1,
 68     postParams: 1,
 69     cacheKey: 1,
 70     cacheTime: 1,
 71     before: 1,
 72     after: 1
 73 };
 74 var GetOptions = function(obj) {
 75     var r = {};
 76     for (var p in obj) {
 77         if (!WhiteList[p]) {
 78             r[p] = obj[p];
 79         }
 80     }
 81     return r;

 83 var WrapDone = function(fn, model, idx) {
 84     return function() {
 85         return fn.apply(model, [model, idx].concat(Slice.call(arguments)));
 86     }
 87 };
 88 var UsedModel = function(m, f) {
 89     if (f) {
 90         for (var i = m.length - 1; i > -1; i--) {
 91             UsedModel(m[i]);
 92         }
 93     } else {
 94         var mm = m.$mm;
 95         if (!m.fromCache && mm.used > 0) {
 96             m.fromCache = true;
 97         }
 98         mm.used++;
 99     }

153     fetchModels: function(models, done, flag) {

168         if (!Magix.isArray(models)) {

170         }

172         var current = 0;
173         var errorMsg;
174         var hasError;
175 
176         var doneArr = new Array(total);
177         var doneArgs = [];
178         var errorArgs = {};
179         var orderlyArr = [];
180 
181         var doneIsArray = Magix.isArray(done);
182         if (doneIsArray) {
183             doneArgs = new Array(done.length);
184         }
185         var doneFn = function(model, idx, data, err) {
186             if (me.$destroy) return; //销毁，啥也不做
187             current++;
188             delete reqModels[model.id];
189             var mm = model.$mm;
190             var cacheKey = mm.cacheKey;
191             doneArr[idx] = model;
192             if (err) {
193                 hasError = true;
194                 errorMsg = err;
195                 errorArgs[idx] = data;
196             } else {
197                 if (!cacheKey || (cacheKey && !modelsCache.has(cacheKey))) {
198                     if (cacheKey) {
199                         modelsCache.set(cacheKey, model);
200                     }
201                     mm.doneAt = Now();
202                     var after = mm.after;
203                     var meta = mm.meta;
204 
205                     if (after) { //有after
206                         SafeExec(after, [model, meta]);
207                     }
208                     host.fireAfter(meta.name, [model]);
209                 }
210             }
211 
212             if (flag == FetchFlags.ONE) { //如果是其中一个成功，则每次成功回调一次
213                 var m = doneIsArray ? done[idx] : done;
214                 if (m) {
215                     UsedModel(model);
216                     doneArgs[idx] = SafeExec(m, [model, hasError ? {
217                         msg: errorMsg
218                     } : null, hasError ? errorArgs : null], me);
219                 }
220             } else if (flag == FetchFlags.ORDER) {
221                 //var m=doneIsArray?done[idx]:done;
222                 orderlyArr[idx] = {
223                     m: model,
224                     e: hasError,
225                     s: errorMsg
226                 };
227                 //
228                 for (var i = orderlyArr.i || 0, t, d; t = orderlyArr[i]; i++) {
229                     d = doneIsArray ? done[i] : done;
230                     UsedModel(t.m);
231                     doneArgs[i] = SafeExec(d, [t.m, t.e ? {
232                         msg: t.s
233                     } : null, orderlyArr.e ? errorArgs : null, doneArgs], me);
234                     if (t.e) {
235                         errorArgs[i] = t.s;
236                         orderlyArr.e = 1;
237                     }
238                 }
239                 orderlyArr.i = i;
240             }
241 
242 
243             if (current >= total) {
244                 errorArgs.msg = errorMsg;
245                 var last = hasError ? errorArgs : null;
246                 if (flag == FetchFlags.ALL) {
247                     UsedModel(doneArr, 1);
248                     doneArr.push(last);
249                     doneArgs[0] = SafeExec(done, doneArr, me);
250                     doneArgs[1] = last;
251                 } else {
252                     doneArgs.push(last);
253                 }
254                 me.$ntId = setTimeout(function() { //前面的任务可能从缓存中来，执行很快
255                     me.doNext(doneArgs);
256                 }, 30);
257             }
258 
259             if (cacheKey && Has(modelsCacheKeys, cacheKey)) {
260                 var fns = modelsCacheKeys[cacheKey].q;
261                 delete modelsCacheKeys[cacheKey];
262                 SafeExec(fns, [data, err], model);
263             }
264 
265         };
266         //
267 
268         for (var i = 0, model; i < models.length; i++) {
269             model = models[i];
270             if (model) {
271                 var modelEntity, modelInfo;
272                 var modelInfo = host.getModel(model);
273                 var cacheKey = modelInfo.cacheKey;
274                 modelEntity = modelInfo.entity;
275                 var wrapDoneFn = WrapDone(doneFn, modelEntity, i);
276 
277                 if (cacheKey && Has(modelsCacheKeys, cacheKey)) {
278                     modelsCacheKeys[cacheKey].q.push(wrapDoneFn);
279                 } else {
280                     if (modelInfo.needUpdate) {
281                         reqModels[modelEntity.id] = modelEntity;
282                         if (cacheKey) {
283                             modelsCacheKeys[cacheKey] = {
284                                 q: [],
285                                 e: modelEntity
286                             };
287                         }
288                         modelEntity.request(wrapDoneFn);
289                     } else {
290                         wrapDoneFn();
291                     }
292                 }
293             } else {
294                 throw new Error('miss attrs:' + models);
295             }
296         }
297         return me;

389     /**
390      * 中止所有model的请求
391      * 注意：调用该方法后会中止请求，并回调error方法
392      */
393     abort: function() {
394         var me = this;
395         clearTimeout(me.$ntId);
396         var host = me.$host;
397         var reqModels = me.$reqModels;
398         var modelsCacheKeys = host.$mCacheKeys;
399 
400         if (reqModels) {
401             for (var p in reqModels) {
402                 var m = reqModels[p];
403                 var cacheKey = m.$mm.cacheKey;
404                 if (cacheKey && Has(modelsCacheKeys, cacheKey)) {
405                     var fns = modelsCacheKeys[cacheKey];
406                     delete modelsCacheKeys[cacheKey];
407                     SafeExec(fns, [true, m, 'aborted'], m);
408                 }
409                 m.abort();
410             }
411         }
412         me.$reqModels = {};
413         me.$queue = [];
414         me.$doTask = false;
415     },
416     /**
417      * 前一个fetchX或saveX任务做完后的下一个任务
418      * @param  {Function} fn 回调
419      * @return {MRequest}
420      */
421     next: function(fn) {
422         var me = this;
423         if (!me.$queue) me.$queue = [];
424         me.$queue.push(fn);
425         if (!me.$doTask) {
426             var args = me.$latest || [];
427             me.doNext.apply(me, [me].concat(args));
428         }
429         return me;
430     },
431     /**
432      * 做下一个任务
433      * @private
434      */
435     doNext: function(preArgs) {
436         var me = this;
437         me.$doTask = false;
438         var queue = me.$queue;
439         if (queue) {
440             var one = queue.shift();
441             if (one) {
442                 SafeExec(one, [me].concat(preArgs), me);
443             }
444         }
445         me.$latest = preArgs;
446     },

449      */
450     destroy: function() {
451         var me = this;
452         me.$destroy = true;
453         me.abort();
454     }

888     getCachedModel: function(modelAttrs) {
889         var me = this;
890         var modelsCache = me.$mCache;
891         var entity = null;
892         var cacheKey;
893         var meta;
894         if (Magix.isString(modelAttrs)) {
895             cacheKey = modelAttrs;
896         } else {
897             meta = me.getModelMeta(modelAttrs);
898             cacheKey = modelAttrs.cacheKey || meta.cacheKey;
899             if (Magix.isFunction(cacheKey)) {
900                 cacheKey = SafeExec(cacheKey, [meta, modelAttrs]);
901             }
902         }
903 
904         if (cacheKey) {
905             var requestCacheKeys = me.$mCacheKeys;
906             var info = requestCacheKeys[cacheKey];
907             if (info) {
908                 entity = info.e;
909             } else if (entity = modelsCache.get(cacheKey)) { //缓存
910                 if (!meta) meta = entity.$mm.meta;
911                 var cacheTime = modelAttrs.cacheTime || meta.cacheTime || 0;
912 
913                 if (Magix.isFunction(cacheTime)) {
914                     cacheTime = SafeExec(cacheTime, [meta, modelAttrs]);
915                 }
916 
917                 if (cacheTime > 0) {
918                     if (Now() - entity.$mm.doneAt > cacheTime) {
919                         me.clearCacheByKey(cacheKey);
920                         entity = null;
921                     }
922                 }
923             }
924         }
925         return entity;
