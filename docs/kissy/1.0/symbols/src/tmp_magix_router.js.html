276     /**
277      * 获取2个location对象之间的差异部分
278      * @param  {Object} oldLocation 原始的location对象
279      * @param  {Object} newLocation 当前的location对象
280      * @return {Object} 返回包含差异信息的对象

283     getChged: function(oldLocation, newLocation) {
284         var oKey = oldLocation.href;
285         var nKey = newLocation.href;
286         var tKey = oKey + '\n' + nKey;
287         var result = ChgdCache.get(tKey);
288         if (!result) {
289             tKey = nKey + '\n' + tKey;
290             result = ChgdCache.get(tKey);
291         }
292         if (!result) {
293             var hasChanged;
294             result = {
295                 params: {}
296             };
297             if (oldLocation[PATHNAME] != newLocation[PATHNAME]) {
298                 result[PATHNAME] = 1;
299                 hasChanged = 1;
300             }
301             if (oldLocation.view != newLocation.view) {
302                 result.view = 1;
303                 hasChanged = 1;
304             }
305             var oldParams = oldLocation[PARAMS],
306                 newParams = newLocation[PARAMS];
307             var p;
308             for (p in oldParams) {
309                 if (oldParams[p] != newParams[p]) {
310                     hasChanged = 1;
311                     result[PARAMS][p] = 1;
312                 }
313             }
314 
315             for (p in newParams) {
316                 if (oldParams[p] != newParams[p]) {
317                     hasChanged = 1;
318                     result[PARAMS][p] = 1;
319                 }
320             }
321             result.occur = hasChanged;
322             result.isParam = isParam;
323             result.isPathname = isPathname;
324             result.isView = isView;
325             ChgdCache.set(tKey, result);
326         }
327         return result;

387     navigate: function(pn, params, replace) {
388         var me = Router;
389 
390         if (!params && Magix.isObject(pn)) {
391             params = pn;
392             pn = EMPTY;
393         }
394         if (params) {
395             pn = Magix.objectToPath({
396                 params: params,
397                 pathname: pn
398             }, IsUtf8)
399         }
400         //TLoc引用
401         //pathObj引用
402         //
403         //temp={params:{},pathname:{}}
404         //
405         //Mix(temp,TLoc,temp);
406         //
407         //
408 
409         if (pn) {
410 
411             var pathObj = Path(pn);
412             var temp = {};
413             temp[PARAMS] = Mix({}, pathObj[PARAMS]);
414             temp[PATHNAME] = pathObj[PATHNAME];
415 
416             if (temp[PATHNAME]) {
417                 if (HashAsNativeHistory) { //指定使用history state但浏览器不支持，需要把query中的存在的参数以空格替换掉
418                     var query = TLoc.query;
419                     if (query && (query = query[PARAMS])) {
420                         for (var p in query) {
421                             if (Has(query, p) && !Has(temp[PARAMS], p)) {
422                                 temp[PARAMS][p] = EMPTY;
423                             }
424                         }
425                     }
426                 }
427             } else {
428                 var ps = Mix({}, TLoc[PARAMS]);
429                 temp[PARAMS] = Mix(ps, temp[PARAMS]);
430                 temp[PATHNAME] = TLoc[PATHNAME];
431             }
432             var tempPath = Magix.objectToPath(temp);
433 
434             var navigate;
435 
436             if (SupportState) {
437                 navigate = tempPath != TLoc.srcQuery;
438             } else {
439                 navigate = tempPath != TLoc.srcHash;
440             }
441 
442             if (navigate) {
443 
444                 if (SupportState) { //如果使用pushState
445                     me.poped = 1;
446                     history[replace ? 'replaceState' : 'pushState'](null, null, tempPath);
447                     me.route();
448                 } else {
449                     Mix(temp, TLoc, temp);
450                     temp.srcHash = tempPath;
451                     temp.hash = {
452                         params: temp[PARAMS],
453                         pathname: temp[PATHNAME]
454                     };
455                     /*
456                         window.onhashchange=function(e){
457                         };
458                         (function(){
459                             location.hash='a';
460                             location.hash='b';
461                             location.hash='c';
462                         }());
463 
464 
465                      */
466                     me.fire('!ul', {
467                         loc: TLoc = temp
468                     }); //hack 更新view的location属性
469                     tempPath = '#!' + tempPath;
470                     if (replace) {
471                         location.replace(tempPath);
472                     } else {
473                         location.hash = tempPath;
474                     }
475                 }
476             }
477         }
