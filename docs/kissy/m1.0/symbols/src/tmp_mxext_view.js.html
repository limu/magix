 56 /**
 57  * @name MxView
 58  * @namespace
 59  * @requires View
 60  * @augments View
 61  */
 62 var MxView = View.extend({
 63     /**
 64      * @lends MxView#
 65      */
 66     /**
 67      * 当前view实例化后调用，供子类重写
 68      * @function
 69      */
 70     mxViewCtor: Magix.noop, //供扩展用
 71     /**
 72      * 调用magix/router的navigate方法
 73      */
 74     navigate: function() {
 75         Router.navigate.apply(Router, arguments);
 76     },
 77     /**
 78      * 让view帮你管理资源，<b>强烈建议对组件等进行托管</b>
 79      * @param {String|Object} key 托管的资源或要共享的资源标识key
 80      * @param {Object} res 要托管的资源
 81      * @return {Object} 返回传入的资源，对于函数会自动进行一次包装
 82      * @example
 83      * init:function(){
 84      *      this.manage('user_list',[//管理对象资源
 85      *          {id:1,name:'a'},
 86      *          {id:2,name:'b'}
 87      *      ]);
 88      * },
 89      * render:function(){
 90      *      var _self=this;
 91      *      var m=new Model();
 92      *      m.load({
 93      *          success:function(resp){
 94      *              //TODO
 95      *              var brix=new BrixDropdownList();
 96      *
 97      *              _self.manage(brix);//管理组件
 98      *
 99      *              var pagination=_self.manage(new BrixPagination());//也可以这样
100      *
101      *              var timer=_self.manage(setTimeout(function(){
102      *                  S.log('timer');
103      *              },2000));//也可以管理定时器
104      *
105      *
106      *              var userList=_self.getManaged('user_list');//通过key取托管的资源
107      *
108      *              S.log(userList);
109      *          },
110      *          error:function(msg){
111      *              //TODO
112      *          }
113      *      });
114      *
115      *      _self.manage(m);
116      * }
117      */
118     manage: function(key, res) {
119         var me = this;
120         var args = arguments;
121         var hasKey = true;
122         if (args.length == 1) {
123             res = key;
124             key = 'res_' + (ResCounter++);
125             hasKey = false;
126         }
127         if (!me.$res) me.$res = {};
128         var destroy;
129         if (Magix.isNumber(res)) {
130             destroy = DestroyTimer;
131         } else if (res && res.destroy) {
132             destroy = Destroy;
133         }
134         var wrapObj = {
135             hasKey: hasKey,
136             res: res,
137             destroy: destroy
138         };
139         me.$res[key] = wrapObj;
140         return res;
141     },
142     /**
143      * 获取托管的资源
144      * @param {String} key 托管资源时传入的标识key
145      * @return {Object}
146      */
147     getManaged: function(key) {
148         var me = this;
149         var cache = me.$res;
150         var sign = me.sign;
151         if (cache && Has(cache, key)) {
152             var wrapObj = cache[key];
153             var resource = wrapObj.res;
154             return resource;
155         }
156         return null;
157     },
158     /**
159      * 移除托管的资源
160      * @param {String|Object} param 托管时标识key或托管的对象
161      * @return {Object} 返回移除的资源
162      */
163     removeManaged: function(param) {
164         var me = this,
165             res = null;
166         var cache = me.$res;
167         if (cache) {
168             if (Has(cache, param)) {
169                 res = cache[param].res;
170                 delete cache[param];
171             } else {
172                 for (var p in cache) {
173                     if (cache[p].res === param) {
174                         res = cache[p].res;
175                         delete cache[p];
176                         break;
177                     }
178                 }
179             }
180         }
181         return res;
182     },
183     /**
184      * 销毁托管的资源
185      * @private
186      */
187     destroyManaged: function(e) {
188         var me = this;
189         var cache = me.$res;
190         //
191         if (cache) {
192             for (var p in cache) {
193                 var o = cache[p];
194                 //var processed=false;
195                 var res = o.res;
196                 var destroy = o.destroy;
197                 var processed = false;
198                 if (destroy) {
199                     destroy(res);
200                     processed = true;
201                 }
202                 if (!o.hasKey) { //如果托管时没有给key值，则表示这是一个不会在其它方法内共享托管的资源，view刷新时可以删除掉
203                     delete cache[p];
204                 }
205                 me.fire('destroyManaged', {
206                     resource: res,
207                     processed: processed
208                 });
209             }
210             if (e.type == 'destroy') { //如果不是刷新，则是view的销毁
211                 //me.un('destroyResource');
212                 delete me.$res;
213             }
214         }
215     },

248     /**
249      * @private
250      */
251     destroyMRequest: function() {
252         var me = this;
253         var cache = me.$res;
254         if (cache) {
255             for (var p in cache) {
256                 var o = cache[p];
257                 var res = o.res;
258                 if (res && res.fetchOne && res.fetchAll) { //销毁MRequest
259                     res.destroy();
260                     delete cache[p];
261                 }
262             }
263         }
264     }
265 }, function() {
266     var me = this;
267     me.on('interact', function() {
268         me.on('rendercall', me.destroyMRequest);
269         me.on('prerender', me.destroyManaged);
270         me.on('destroy', me.destroyManaged);
271     });
272     me.mxViewCtor();
273 });

275 /**
276  * view销毁托管资源时发生
277  * @name MxView#destroyResource
278  * @event
279  * @param {Object} e
280  * @param {Object} e.resource 托管的资源
281  * @param {Boolean} e.processed 表示view是否对这个资源做过销毁处理，目前view仅对带 abort destroy dispose方法的资源进行自动处理，其它资源需要您响应该事件自已处理
282  */
283     return MxView;
284 }, {
285     requires: ["magix/magix", "magix/view", "magix/router"]
