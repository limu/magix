122      * @lends View
123      */
124     prepare: function(oView) {
125         var me = this;
126         if (!oView.extend) {
127             oView.extend = me.extend;
128         }
129         if (!oView[WrapKey]) { //只处理一次
130             oView[WrapKey] = 1;
131             var prop = oView.prototype;
132             var old, temp, name, evts, idx, revts = {};
133             for (var p in prop) {
134                 old = prop[p];
135                 temp = p.match(EvtMethodReg);
136                 if (temp) {
137                     name = temp[1];
138                     evts = temp[2];
139                     prop[name] = old;
140                     evts = evts.split(COMMA);
141                     for (idx = evts.length - 1; idx > -1; idx--) {
142                         temp = evts[idx];
143                         revts[temp] = 1;
144                         prop[name + MxEvtSplit + temp] = old;
145                     }
146                 } else if (WrapAsynUpdateNames[old]) {
147                     prop[p] = WrapFn(old);
148                 }
149             }
150             prop.$evts = revts;
151         }
152     }

821     var Tmpls = {}, Locker = {};
822     View.prototype.fetchTmpl = function(fn) {
823         var me = this;
824         var hasTemplate = 'template' in me;
825         if (!hasTemplate) {
826             if (Has(Tmpls, me.path)) {
827                 fn(Tmpls[me.path]);
828             } else {
829                 var file = AppRoot + me.path + '.html';
830                 var l = Locker[file];
831                 var onload = function(tmpl) {
832                     fn(Tmpls[me.path] = tmpl);
833                 };
834                 if (l) {
835                     l.push(onload);
836                 } else {
837                     l = Locker[file] = [onload];
838                     IO({
839                         url: file + Suffix,
840                         success: function(x) {
841                             SafeExec(l, x);
842                             delete Locker[file];
843                         },
844                         error: function(e, m) {
845                             SafeExec(l, m);
846                             delete Locker[file];
847                         }
848                     });
849                 }
850             }
851         } else {
852             fn(tmpl);
853         }
854     };
