        ],function(err,model){//m1,m2,m3，谁快先调用谁，且被调用三次

        var r=MM.fetchOrder([
            {name:'M1'},

            {name:'M3'}
        ],function(err,model){//m1什么时间返回，该回调什么时间被调用
            if(err){
                alert(err.msg);
            }else{
                alert(model.get('name'));
            }
        },function(err,model){//m2什么时间返回，该回调什么时间被调用
            if(err){
                alert(err.msg);
            }else{
                alert(model.get('name'));
            }
        });
  • fetchOrder (models, done)
    获取models，按顺序回回调done
    Parameters
    •   models {String|Array} 获取models时的描述信息，如:{name:'Home',cacheKey:'key',urlParams:{a:'12'},postParams:{b:2}}
    •   done {Function} 完成时的回调
    Returns
    • {MRequest}
    Example
            //代码片断：
        //1：当按顺序获取多个model，回调只有一个时
        var r=MM.fetchOrder([
            {name:'M1'},
            {name:'M2'},
            {name:'M3'}
        ],function(err,model){//回调按M1,M2,M3的顺序被调用3次
            if(err){
                alert(err.msg);
            }else{
                alert(model.get('name'));
            }
        });

        //2:当按顺序获取多个model，回调多于一个时
        var r=MM.fetchOrder([
            {name:'M1'},
            {name:'M2'},
            {name:'M3'}
        ],function(err,model){//首先被调用
            if(err){
                alert(err.msg);
            }else{
                alert(model.get('name'));
            }
        },function(err,model){//其次被调用
            if(err){
                alert(err.msg);
            }else{
                alert(model.get('name'));
            }
        });

  • registerModels (models)
    注册APP中用到的model
    Parameters
    •   models {Object|Array} 模块描述信息
    •   models.name {String} app中model的唯一标识
    •   models.options {Object} 传递的参数信息，如{uri:'test',isJSONP:true,updateIdent:true}
    •   models.urlParams {Object} 发起请求时，默认的get参数对象
    •   models.postParams {Object} 发起请求时，默认的post参数对象
    •   models.cacheKey {String} 指定model缓存的key，当指定后，该model会进行缓存，下次不再发起请求
    •   models.cacheTime {Integer} 缓存过期时间，以毫秒为单位，当过期后，再次使用该model时会发起新的请求(前提是该model指定cacheKey被缓存后cacheTime才有效)
    •   models.before {Function} model在发起请求前的回调
    •   models.after {Function} model在结束请求，并且成功后回调
