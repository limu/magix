1 /**
  2  * @fileOverview 多播事件对象
  3  * @author 行列<xinglie.lkf@taobao.com>
  4  * @version 1.0
  5  **/
  6 define("magix/event", ["magix/magix"], function(Magix) {
  7     /**
  8  * 根据名称生成事件数组的key
  9  * @param {Strig} name 事件名称
 10  * @return {String} 包装后的key
 11  */
 12 var GenKey = function(name) {
 13     return '~' + name;
 14 };
 15 
 16 var SafeExec = Magix.safeExec;
 17 /**
 18  * 多播事件对象
 19  * @name Event
 20  * @namespace
 21  */
 22 var Event = {
 23     /**
 24      * @lends Event
 25      */
 26     /**
 27      * 触发事件
 28      * @param {String} name 事件名称
 29      * @param {Object} data 事件对象
 30      * @param {Boolean} remove 事件触发完成后是否移除这个事件的所有监听
 31      * @param {Boolean} lastToFirst 是否从后向前触发事件的监听列表
 32      */
 33     fire: function(name, data, remove, lastToFirst) {
 34         var key = GenKey(name),
 35             me = this,
 36             list = me[key];
 37         if (list) {
 38             if (!data) data = {};
 39             if (!data.type) data.type = name;
 40             var end = list.length,
 41                 len = end - 1,
 42                 idx, t;
 43             while (end--) {
 44                 idx = lastToFirst ? end : len - end;
 45                 t = list[idx];
 46                 if (t.d || t.r) {
 47                     list.splice(idx, 1);
 48                     len--;
 49                 }
 50                 if (!t.d) SafeExec(t.f, data, me);
 51             }
 52         }
 53         if (remove) {
 54             delete me[key];
 55         }
 56     },
 57     /**
 58      * 绑定事件
 59      * @param {String} name 事件名称
 60      * @param {Function} fn 事件回调
 61      * @param {Interger|Boolean} insertOrRemove 事件监听插入的位置或触发后是否移除监听
 62      * @example
 63      * var T=Magix.mix({},Event);
 64      * T.on('done',function(e){
 65      *
 66      * });
 67      *
 68      * T.on('done',function(e){
 69      *
 70      * },0)//监听插入到开始位置
 71      *
 72      * T.on('done',function(e){
 73      *
 74      * },true)//触发后即删除该监听
 75      *
 76      * T.fire('done',{
 77      *     data:'test'
 78      * })
 79      */
 80     on: function(name, fn, insertOrRemove) {
 81         var key = GenKey(name);
 82         var list = this[key] || (this[key] = []);
 83         if (Magix.isNumeric(insertOrRemove)) {
 84             list.splice(insertOrRemove, 0, {
 85                 f: fn
 86             });
 87         } else {
 88             list.push({
 89                 f: fn,
 90                 r: insertOrRemove
 91             });
 92         }
 93     },
 94     /**
 95      * 解除事件绑定
 96      * @param {String} name 事件名称
 97      * @param {Function} fn 事件回调
 98      */
 99     un: function(name, fn) {
100         var key = GenKey(name),
101             list = this[key];
102         if (list) {
103             if (fn) {
104                 for (var i = list.length - 1, t; i >= 0; i--) {
105                     t = list[i];
106                     if (t.f == fn && !t.d) {
107                         t.d = 1;
108                         break;
109                     }
110                 }
111             } else {
112                 delete this[key];
113             }
114         }
115     }
116 };
117     return Event;
118 });
