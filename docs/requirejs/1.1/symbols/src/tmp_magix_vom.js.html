1 /**
  2  * @fileOverview VOM
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 define("magix/vom", ["magix/vframe", "magix/magix", "magix/event"], function(Vframe, Magix, Event) {
  7     var Has = Magix.has;
  8 var Mix = Magix.mix;
  9 var VframesCount = 0;
 10 var FirstVframesLoaded = 0;
 11 var LastPercent = 0;
 12 var FirstReady = 0;
 13 var Vframes = {};
 14 var Loc = {};
 15 
 16 /**
 17  * VOM对象
 18  * @name VOM
 19  * @namespace
 20  */
 21 var VOM = Magix.mix({
 22     /**
 23      * @lends VOM
 24      */
 25     /**
 26      * 获取所有的vframe对象
 27      * @return {Object}
 28      */
 29     all: function() {
 30         return Vframes;
 31     },
 32     /**
 33      * 注册vframe对象
 34      * @param {Vframe} vf Vframe对象
 35      */
 36     add: function(vf) {
 37         if (!Has(Vframes, vf.id)) {
 38             VframesCount++;
 39             Vframes[vf.id] = vf;
 40             vf.owner = VOM;
 41             VOM.fire('add', {
 42                 vframe: vf
 43             });
 44         }
 45     },
 46     /**
 47      * 根据vframe的id获取vframe对象
 48      * @param {String} id vframe的id
 49      * @return {Vframe|null} vframe对象
 50      */
 51     get: function(id) {
 52         return Vframes[id];
 53     },
 54     /**
 55      * 删除已注册的vframe对象
 56      * @param {String} id vframe对象的id
 57      */
 58     remove: function(id) {
 59         var vf = Vframes[id];
 60         if (vf) {
 61             VframesCount--;
 62             if (vf.fcc) FirstVframesLoaded--;
 63             delete Vframes[id];
 64             VOM.fire('remove', {
 65                 vframe: vf
 66             });
 67         }
 68     },
 69     /**
 70      * 通知其中的一个vframe创建完成
 71      * @private
 72      */
 73     vfCreated: function() {
 74         if (!FirstReady) {
 75             FirstVframesLoaded++;
 76             var np = FirstVframesLoaded / VframesCount;
 77             if (LastPercent < np) {
 78                 VOM.fire('progress', {
 79                     percent: LastPercent = np
 80                 }, FirstReady = (np == 1));
 81             }
 82         }
 83     },
 84     /**
 85      * 获取根vframe对象
 86      */
 87     root: function() {
 88         return Vframe.root(VOM, Loc);
 89     },
 90     /**
 91      * 向vframe通知地址栏发生变化
 92      * @param {Object} e 事件对象
 93      * @param {Object} e.location window.location.href解析出来的对象
 94      * @param {Object} e.changed 包含有哪些变化的对象
 95      * @private
 96      */
 97     locChged: function(e) {
 98         var loc = e.loc;
 99         var hack;
100         if (loc) {
101             hack = 1;
102         } else {
103             loc = e.location;
104         }
105         Mix(Loc, loc);
106         if (!hack) {
107             var vf = VOM.root();
108             var chged = e.changed;
109             if (chged.isView()) {
110                 vf.mountView(loc.view);
111             } else {
112                 vf.locChged(loc, chged);
113             }
114         }
115     }
116     /**
117      * view加载完成进度
118      * @name VOM.progress
119      * @event
120      * @param {Object} e
121      * @param {Object} e.precent 百分比
122      */
123     /**
124      * 注册vframe对象时触发
125      * @name VOM.add
126      * @event
127      * @param {Object} e
128      * @param {Vframe} e.vframe
129      */
130     /**
131      * 删除vframe对象时触发
132      * @name VOM.remove
133      * @event
134      * @param {Object} e
135      * @param {Vframe} e.vframe
136      */
137 }, Event);
138     return VOM;
139 });
