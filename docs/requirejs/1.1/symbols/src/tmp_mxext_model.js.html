138     getParams: function(type) {
139         var me = this;
140         if (!type) {
141             type = Model.GET;
142         } else {
143             type = type.toUpperCase();
144         }
145         var k = '$' + type;
146         var params = me[k];
147         var arr = [];
148         var v;
149         if (params) {
150             for (var p in params) {
151                 v = params[p];
152                 if (Magix.isArray(v)) {
153                     for (var i = 0; i < v.length; i++) {
154                         arr.push(p + '=' + Encode(v[i]));
155                     }
156                 } else {
157                     arr.push(p + '=' + Encode(v));
158                 }
159             }
160         }
161         return arr.join('&');

199         if (!me[k]) me[k] = {};

201             for (var p in obj1) {
202                 if (!ignoreIfExist || !me[k][p]) {
203                     me[k][p] = obj1[p];
204                 }
205             }
206         } else if (obj1) {
207             if (!ignoreIfExist || !me[k][obj1]) {
208                 me[k][obj1] = obj2;

317     set: function(key, val, saveKeyList) {
318         var me = this;
319         if (!me.$attrs) me.$attrs = {};
320         if (saveKeyList && !me.$keys) {
321             me.$keys = [];
322         }
323         if (Magix.isObject(key)) {
324             if (!Magix.isObject(val)) {
325                 val = {}
326             }
327             for (var p in key) {
328                 if (saveKeyList) {
329                     me.$keys.push(p);
330                 }
331                 if (!Magix.has(val, p)) {
332                     me.$attrs[p] = key[p];
333                 }
334             }
335         } else if (key) {
336             if (saveKeyList) {
337                 me.$keys.push(key);
338             }
339             me.$attrs[key] = val;
340         }

342     /**
343      * 向服务器请求，加载或保存数据
344      * @param {Function} callback 请求成功或失败的回调
345      */
346     request: function(callback, options) {
347         if (!callback) callback = function() {};
348         var me = this;
349         me.$abort = false;
350         var temp = function(err, data) {
351             if (!me.$abort) {
352                 if (err) {
353                     callback(err, data, options);
354                 } else {
355                     if (data) {
356                         var val = me.parse(data);
357                         if (!Magix.isObject(val)) {
358                             val = {
359                                 data: val
360                             };
361                         }
362                         me.set(val, null, true);
363                     }
364                     callback(err, data, options);
365                 }
366             }
367         };
368 
369 
370 
371 
372 
373 
374         me.$trans = me.sync(temp, options);
375     },
