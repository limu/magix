  6 (function(W) {
  7     var noop = function() {};
  8     if (!W.console) {
  9         W.console = {
 10             log: noop,
 11             info: noop,
 12             error: noop
 13         }
 14     };
 15     var tempCfg, cCfg = {};
 16     if (!W.Magix) {
 17         W.Magix = {
 18             config: function(cfg) {
 19                 for (var p in cfg) {
 20                     cCfg[p] = cfg[p];
 21                 }
 22             },
 23             start: function(cfg) {
 24                 tempCfg = cfg;
 25             }
 26         };
 27         require(['magix/magix'], function(M) {
 28             W.Magix = M;
 29             M.config(cCfg);
 30             if (tempCfg) {
 31                 M.start(tempCfg);
 32             }
 33         });
 34     }
 35 })(this);
